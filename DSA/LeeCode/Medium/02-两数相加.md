## [02 两数相加](https://leetcode.cn/problems/add-two-numbers/)

### Python3

【单个循环】：
```py
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        head = tail = ListNode()
        carry = 0
        while l1 is not None or l2 is not None:
            a = l1.val if l1 is not None else 0
            b = l2.val if l2 is not None else 0
            sum = a + b + carry
            carry = sum // 10

            tail.next = ListNode(sum % 10)
            tail = tail.next

            l1 = l1.next if l1 is not None else None
            l2 = l2.next if l2 is not None else None

        if carry == 1:
            tail.next = ListNode(carry)

        return head.next
```

【多个循环】：
```py
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        if l1 is None:
            return l2
        if l2 is None:
            return l1
        if l1 is None and l2 is None:
            return ListNode()

        p = head = ListNode()
        carry = 0
        while l1 is not None and l2 is not None:
            sum = l1.val + l2.val + carry
            carry = sum // 10
            p.next = ListNode(sum % 10)
            p, l1, l2 = p.next, l1.next, l2.next

        l = l1 if l1 is not None else l2
        while l is not None:
            sum = l.val + carry
            carry = sum // 10
            p.next = ListNode(sum % 10)
            p, l = p.next, l.next

        if carry == 1:
            p.next = ListNode(1)

        return head.next
```
